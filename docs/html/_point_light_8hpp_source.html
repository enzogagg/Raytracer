100 {

106 if (!setting["color"].isGroup())

108
109 const libconfig::Setting &colorSetting = setting["color"];
110
111 if (!colorSetting.exists("r") || !colorSetting.exists("g") ||
112 !colorSetting.exists("b") || !colorSetting["r"].isNumber() ||
113 !colorSetting["g"].isNumber() || !colorSetting["b"].isNumber())
114 throw std::runtime_error("PointLight: r, g, b is not a number");
115
116 auto getFloatOrInt = [](const libconfig::Setting &s) -> float {
117 if (s.getType() == libconfig::Setting::TypeFloat)
118 return static_cast<float>(s);
119 if (s.getType() == libconfig::Setting::TypeInt)
120 return static_cast<float>(static_cast<int>(s));
121 throw std::runtime_error("PointLight: value is not a number");
